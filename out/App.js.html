function FriendsList() {
  const friends = initialFriends;
  return (
    <ul>
      {friends.map(friend => <Friend friend={friend} key={friend.id} />)}
    </ul>
  );
}

/**
 * This function takes an object called "friend" which contains 1.image 2.name 3.balance 4.id
 * and dispaly them into a list item, which contain conditional rendering for how much balance you have < 0 means
 * the balance is negative and you owe that person that much money ex: {-7}  
 * @param {object} friend object comes from a map function ittraites on friends array
 * @returns {li} returns a Friend component
 */
function Friend({ friend }) {
  return (
    <li>
      <img src={friend.image} alt={friend.name} />
      <h3>{friend.name}</h3>
      {friend.balance < 0 && (
        <p className="red">
          You Owe {friend.name} {Math.abs(friend.balance)}$
        </p>
      )}
      {friend.balance > 0 && (
        <p className="green">
          {friend.name} owes you {Math.abs(friend.balance)}$
        </p>
      )}
      {friend.balance === 0 && (
        <p>
          You and {friend.name} are even
        </p>
      )}
      <button className="button">
        Select
      </button>
    </li>
  )
}
